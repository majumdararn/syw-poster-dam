\begin{textblock}{86}(7.5,43)
\SECTION{Showyourwork philosophy for reproducing scientific article}
\begin{tikzpicture}
	\anode{0}{0}{74cm}{
    \emph{Showyourwork!}~\cite{Luger2021} is an open-source reproducibility framework designed to enable transparent and fully reproducible scientific research. It organizes analysis scripts, software dependencies, and the manuscript within a GitHub repository, allowing any independent researcher to clone the project and regenerate the complete article. The manuscript preparation can be integrated with Overleaf, enabling collaborative writing and seamless synchronization between analysis and documentation.

    Once reproduced, figures can be modified directly by editing the underlying scripts that generate them. Raw datasets are typically accessed from a Zenodo repository with read-only permissions, while intermediate and processed data products can be automatically uploaded to a separate Zenodo repository with read–write access.

    \begin{figure}
    {
        \script{overview.py}
        \begin{center}
        \includegraphics[width=40cm, trim=0cm 1.3cm 0cm 1cm, clip]{figures/overview.pdf}
        \caption{Framework of \emph{showyourwork!}: GitHub hosts analysis scripts, environment files, dependency rules, and the manuscript; Zenodo stores datasets; and Overleaf enables manuscript editing. Commits trigger automated figure generation and manuscript compilation via GitHub Actions, producing a final version and an arXiv-ready tarball. \label{fig:fig_flw_chart}}    
        \end{center}
    }
    \end{figure}
	}
\end{tikzpicture}
\end{textblock}
